      const scrollContainer = document.querySelector(".scroll-container");

      scrollContainer.addEventListener("scroll", () => {
        console.log("滚动");
        clearTimeout(timer);
        if (isScroll) {
          scrollContainer.classList.remove("show-scrollbar");
        }
        isScroll = false;
        timer = setTimeout(() => {
          console.log("隐藏滚动条");

        }, 2000);
