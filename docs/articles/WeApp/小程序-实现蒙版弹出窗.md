js:

```js
Page({
  data: {
    showModal: false,
  },
  onLoad(o) {},
  showMask: function () {
    this.setData({
      showModal: true,
    });
  },
  hideMask: function () {
    this.setData({
      showModal: false,
    });
  },
});
```
